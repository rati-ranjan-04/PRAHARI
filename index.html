-- ═══════════════════════════════════════════════════════════ -- PRAHARI v5 · PostGIS Geospatial Incident Schema -- Requires: PostgreSQL 14+ with PostGIS 3.x extension -- ═══════════════════════════════════════════════════════════ CREATE EXTENSION IF NOT EXISTS postgis; CREATE EXTENSION IF NOT EXISTS "uuid-ossp"; CREATE TABLE geospatial_incidents ( id UUID PRIMARY KEY DEFAULT uuid_generate_v4(), incident_type VARCHAR(50) NOT NULL, -- 'DDoS','Malware','IoT_Alert','Ransomware' severity VARCHAR(20) NOT NULL, -- 'CRITICAL','HIGH','MEDIUM','LOW' source_location GEOGRAPHY(POINT,4326), -- attacker origin target_location GEOGRAPHY(POINT,4326), -- attacked target source_ip INET, target_ip INET, source_country CHAR(2), target_country CHAR(2), source_city VARCHAR(100), target_city VARCHAR(100), created_at TIMESTAMPTZ DEFAULT NOW(), expires_at TIMESTAMPTZ, -- TTL expiry (created_at + 5min) is_active BOOLEAN DEFAULT TRUE, payload_bytes BIGINT, confidence SMALLINT CHECK (confidence BETWEEN 0 AND 100), sensor_id UUID REFERENCES iot_sensors(id) ON DELETE SET NULL, metadata JSONB, ttl_seconds INT DEFAULT 300 ); CREATE OR REPLACE FUNCTION set_incident_ttl() RETURNS TRIGGER AS $$ BEGIN NEW.expires_at := NEW.created_at + (NEW.ttl_seconds * INTERVAL '1 second'); RETURN NEW; END; $$ LANGUAGE plpgsql; CREATE TRIGGER trg_incident_ttl BEFORE INSERT ON geospatial_incidents FOR EACH ROW EXECUTE FUNCTION set_incident_ttl();
-- ═══════════════════════════════════════════════════════════ -- IoT Sensor Registry — Motion, Thermal, Acoustic, Network -- ═══════════════════════════════════════════════════════════ CREATE TABLE iot_sensors ( id UUID PRIMARY KEY DEFAULT uuid_generate_v4(), sensor_name VARCHAR(100) NOT NULL UNIQUE, sensor_type VARCHAR(50) NOT NULL, location GEOGRAPHY(POINT,4326) NOT NULL, city VARCHAR(100), country CHAR(2), status VARCHAR(20) DEFAULT 'active', last_alert TIMESTAMPTZ, alert_count INT DEFAULT 0, ttl_seconds INT DEFAULT 300, alert_expires TIMESTAMPTZ, sensitivity SMALLINT DEFAULT 75, firmware_ver VARCHAR(20), created_at TIMESTAMPTZ DEFAULT NOW(), updated_at TIMESTAMPTZ DEFAULT NOW() ); CREATE OR REPLACE FUNCTION update_sensor_alert() RETURNS TRIGGER AS $$ BEGIN IF NEW.status = 'triggered' AND OLD.status != 'triggered' THEN NEW.last_alert := NOW(); NEW.alert_expires := NOW() + (NEW.ttl_seconds * INTERVAL '1 second'); NEW.alert_count := NEW.alert_count + 1; END IF; NEW.updated_at := NOW(); RETURN NEW; END; $$ LANGUAGE plpgsql;
-- ═══════════════════════════════════════════════════════════ -- Spatial Indexes + TTL Cleanup Functions + pg_cron -- ═══════════════════════════════════════════════════════════ CREATE INDEX idx_incidents_source ON geospatial_incidents USING GIST(source_location); CREATE INDEX idx_incidents_target ON geospatial_incidents USING GIST(target_location); CREATE INDEX idx_sensors_location ON iot_sensors USING GIST(location); CREATE INDEX idx_incidents_active ON geospatial_incidents(is_active, expires_at) WHERE is_active = TRUE; CREATE OR REPLACE FUNCTION cleanup_expired_incidents() RETURNS INT AS $$ DECLARE expired_count INT; BEGIN UPDATE geospatial_incidents SET is_active = FALSE WHERE expires_at < NOW() AND is_active = TRUE; GET DIAGNOSTICS expired_count = ROW_COUNT; RETURN expired_count; END; $$ LANGUAGE plpgsql; -- Schedule via pg_cron every 30 seconds SELECT cron.schedule('ttl-cleanup', '30 seconds', 'SELECT cleanup_expired_incidents(); SELECT cleanup_expired_sensor_alerts();'); -- Proximity query: incidents within 500km SELECT * FROM geospatial_incidents WHERE ST_DWithin(target_location, ST_MakePoint(77.2, 28.6)::geography, 500000) AND is_active = TRUE ORDER BY created_at DESC LIMIT 50;
-- ═══════════════════════════════════════════════════════════ -- WebSocket Event Store + Node.js Server Architecture -- ═══════════════════════════════════════════════════════════ CREATE TABLE websocket_events ( id BIGSERIAL PRIMARY KEY, event_type VARCHAR(50) NOT NULL, payload JSONB NOT NULL, sensor_id UUID REFERENCES iot_sensors(id), incident_id UUID REFERENCES geospatial_incidents(id), client_id VARCHAR(64), latency_ms INT, created_at TIMESTAMPTZ DEFAULT NOW() ); /* ─── Node.js WebSocket Server (ws + pg) ────────────── const WebSocket = require('ws'); const { Pool } = require('pg'); const pool = new Pool({ connectionString: process.env.DB_URL }); const wss = new WebSocket.Server({ port: 8080 }); pool.connect((err, client) => { client.query('LISTEN sensor_alerts'); client.on('notification', async (msg) => { const payload = JSON.parse(msg.payload); wss.clients.forEach(ws => { if (ws.readyState === WebSocket.OPEN) { ws.send(JSON.stringify({ type: 'sensor_alert', sensor_id: payload.sensor_id, lat: payload.lat, lng: payload.lng, severity: payload.severity, ttl: 300, timestamp: Date.now() })); } }); }); }); // TTL heartbeat every 5s setInterval(() => { wss.clients.forEach(ws => { if (ws.readyState === WebSocket.OPEN) ws.send(JSON.stringify({ type:'ttl_tick', ts: Date.now() })); }); }, 5000); ────────────────────────────────────────────── */ CREATE OR REPLACE FUNCTION notify_sensor_alert() RETURNS TRIGGER AS $$ BEGIN IF NEW.status = 'triggered' THEN PERFORM pg_notify('sensor_alerts', json_build_object( 'sensor_id', NEW.id, 'lat', ST_Y(NEW.location::geometry), 'lng', ST_X(NEW.location::geometry), 'city', NEW.city )::text); END IF; RETURN NEW; END; $$ LANGUAGE plpgsql;
